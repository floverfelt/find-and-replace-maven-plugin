25    /**
26     * Stupid simple implementation of file walking, renaming, etc.
27     *
28     * @param log the maven-plugin log
29     * @param baseDir the directory to start in
30     * @param isRecursive whether to recurse further
31     * @param findRegex the regex to find
32     * @param replaceValue the value to replace the found regex
33     * @param fileMasks the file masks to use
34     * @param exclusions the filename regex to exclude
35     * @param processFileContents whether to process file contents
36     * @param processFilenames whether to process file names
37     * @param processDirectoryNames whether to process directory names
38     */
39    public static void process(Log log, Path baseDir, boolean isRecursive, Pattern findRegex, String replaceValue,
40                               List<String> fileMasks, List<Pattern> exclusions, boolean processFileContents,
41                               boolean processFilenames, boolean processDirectoryNames) throws IOException {
42  
43      // Load in the files in the base dir
44      List<File> filesToProcess = new ArrayList<>(Arrays.asList(Objects.requireNonNull(new File(baseDir.toUri()).listFiles())));
45  
46      ListIterator<File> iterator = filesToProcess.listIterator();
47  
48      while (iterator.hasNext()) {
49  
50        File file = iterator.next();
51  
52        // Remove the file from processing
53        iterator.remove();
54  
55        // Perform dir checks
56        if (file.isDirectory()) {
57          iterator = processDirectory(iterator, isRecursive, file, processDirectoryNames, exclusions, findRegex, replaceValue, log);
58        }
59  
60        if (file.isFile()) {
61          processFile(log, exclusions, file, fileMasks, processFileContents, findRegex, replaceValue, processFilenames);
62        }
63  
64      }
65  
66    }

70    private static ListIterator<File> processDirectory(ListIterator<File> iterator, boolean isRecursive, File file,
71                                             boolean processDirectoryNames, List<Pattern> exclusions, Pattern findRegex,
72                                             String replaceValue, Log log) throws IOException {
73  
74      // Rename the directory
75      if (processDirectoryNames && !shouldExcludeFile(exclusions, file)) {
76          file = renameFile(log, file, findRegex, replaceValue);
77      }
78  
79      // If recursive, add child files to iterator
80      if (isRecursive) {
81        File[] filesToAdd = file.listFiles();
82        assert filesToAdd != null;
83        for (File f : filesToAdd) {
84          iterator.add(f);
85          iterator.previous();
86        }
87      }
88  
89      return iterator;
90  
91    }

94    private static boolean shouldExcludeFile(List<Pattern> exclusions, File file) {
95  
96      for (Pattern p : exclusions) {
97        if (p.matcher(file.getName()).find()) {
98          return true;
99        }
100     }
101 
102     return false;
103 
104   }

106   private static boolean shouldProcessFile(List<String> fileMasks, File file) {
107 
108     if (fileMasks.isEmpty()) {
109       return true;
110     }
111 
112     for (String fileMask : fileMasks) {
113       if (file.getName().endsWith(fileMask)) {
114         return true;
115       }
116     }
117 
118     return false;
119 
120   }

122   private static File renameFile(Log log, File file, Pattern findRegex, String replaceValue) throws IOException {
123 
124     Path filePath = file.toPath();
125     Path parentDir = filePath.getParent();
126     String oldName = file.getName();
127     Matcher m = findRegex.matcher(oldName);
128     String newName = m.replaceAll(replaceValue);
129 
130     if (!newName.equals(oldName)) {
131       Path targetPath = Paths.get(parentDir.toString(), newName);
132 
133       log.info(String.format("Renaming %s to %s", oldName, newName));
134 
135       return new File(Files.move(filePath, targetPath).toUri());
136 
137     }
138 
139     return file;
140 
141   }

143   private static void processFileContents(File file, Pattern findRegex, String replaceValue) throws IOException {
144 
145     File tempFile = File.createTempFile("tmp","tmp", file.getParentFile());
146 
147     try (BufferedReader fileReader = new BufferedReader(new FileReader(file))) {
148       try (FileWriter fileWriter = new FileWriter(tempFile)) {
149 
150         Stream<String> lines = fileReader.lines();
151 
152         lines.forEach(line -> {
153           Matcher m = findRegex.matcher(line);
154           line = m.replaceAll(replaceValue);
155           try {
156             fileWriter.write(line + "\n");
157           } catch (IOException e) {
158             throw new RuntimeException(e);
159           }
160         });
161       }
162     }
163 
164     if (!file.delete()) {
165       throw new IOException("Failed to delete file at: " + file.getPath());
166     }
167 
168     if (!tempFile.renameTo(file)) {
169       throw new IOException("Failed to rename temp file at: " + tempFile.getPath() + " to " + file.getPath());
170     }
171 
172   }

174   private static void processFile(Log log, List<Pattern> exclusions, File file, List<String> fileMasks,
175                                   boolean processFileContents, Pattern findRegex, String replaceValue,
176                                   boolean processFilenames) throws IOException {
177 
178     if (shouldExcludeFile(exclusions, file) || !shouldProcessFile(fileMasks, file)) {
179       return;
180     }
181 
182     if (processFileContents) {
183       processFileContents(file, findRegex, replaceValue);
184     }
185 
186     if (processFilenames) {
187       renameFile(log, file, findRegex, replaceValue);
188     }
189 
190   }
