166   private void setupReplacementTypes() throws MojoExecutionException {
167 
168     String[] replacementTypeList = StringUtils.split(replacementType, ",");
169 
170     String logMessage = "Mode set to ";
171 
172     for (String replacementTypeVal : replacementTypeList) {
173 
174       if (!validReplacementTypes.contains(replacementTypeVal)) {
175         throw new MojoExecutionException("Invalid replacementType specified: " + replacementTypeVal);
176       }
177 
178       if (FILE_CONTENTS.equals(replacementTypeVal)) {
179         getLog().info(logMessage + FILE_CONTENTS);
180         processFileContents = true;
181       }
182 
183       if (FILENAMES.equals(replacementTypeVal)) {
184         getLog().info(logMessage + FILENAMES);
185         processFilenames = true;
186       }
187 
188       if (DIRECTORY_NAMES.equals(replacementTypeVal)) {
189         getLog().info(logMessage + DIRECTORY_NAMES);
190         processDirectoryNames = true;
191       }
192 
193     }
194 
195   }

211     }

215   private void setupBaseDir() {
216 
217     if (project == null) {
218       baseDirPath = Paths.get(baseDir);

220       baseDirPath = Paths.get(baseDir);

222       baseDirPath = Paths.get(project.getBasedir().getAbsolutePath(), baseDir);
