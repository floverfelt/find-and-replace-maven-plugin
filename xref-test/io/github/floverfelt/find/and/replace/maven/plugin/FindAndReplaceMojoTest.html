447   private void setFieldValue(Object obj, String fieldName, Object val)
448       throws NoSuchFieldException, IllegalAccessException {
449     Field field = obj.getClass().getDeclaredField(fieldName);
450     field.setAccessible(true);
451 
452     Field modifiersField = Field.class.getDeclaredField("modifiers");
453     modifiersField.setAccessible(true);
454     modifiersField.setInt(field, field.getModifiers() & ~Modifier.FINAL);
455 
456     field.set(obj, val);
457   }

460

467   private boolean fileContains(File file, String findValue) throws IOException {
468 
469     try (BufferedReader fileReader = new BufferedReader(new FileReader(file))) {
470 
471       Stream<String> lines = fileReader.lines();
472 
473       return lines.anyMatch(line -> line.contains(findValue));
